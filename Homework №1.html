def generate_samples(NumSamples = 100): 
    v = np.random.normal(loc=0, scale=2, size=(NumSamples, 1))
    epsilon = np.random.normal(loc=0, scale=0.1, size=(NumSamples, 1))

def calculate_regression_by_standard_method(v, res):

    # Полиномиальная регрессия (степень полинома = 3)
    model_poly = make_pipeline(PolynomialFeatures(degree=2), LinearRegression())
    model_poly.fit(v, res)
    prediction_poly = model_poly.predict(v)
    mse_poly = np.mean((res - prediction_poly)**2)
    print("Полиномиальная регрессия (степень 2): MSE =", mse_poly)

    # Кубическая сплайновая интерполяция
    model_spline = interp1d(v.squeeze(), res.squeeze(), kind='cubic')
    prediction_spline = np.array([model_spline(v.squeeze())]).T
    #prediction_spline = model_spline(v.squeeze())
    mse_spline = np.mean((res - prediction_spline)**2)
    print("Кубическая сплайновая интерполяция: MSE =", mse_spline)

    return prediction_poly, prediction_spline

Задание 1

Общее уравнение полиномиальной регрессии степени d имеет вид:

y = β₀ + β₁x + β₂x² + ... + βₚxᵈ + ε

где:

  • y - зависимая переменная;
  • x - независимая переменная;
  • β₀, β₁, ..., βₚ - коэффициенты регрессии, которые необходимо оценить;
  • ε - случайная ошибка.

Коэффициенты β можно оценить методом наименьших квадратов (МНК):

β = (X^T X)^-1 X^T y

где:

  • X - матрица полиномиальных признаков размера (n x (d+1)), где n - количество наблюдений, а (d+1) - количество коэффициентов;
  • y - вектор зависимой переменной размера (n x 1).

Матрица X строится следующим образом: X = [1 x x² ... xᵈ]

import matplotlib.pyplot as plt
def polynomial_regression_with_matrics(x, y, degree):
    # Создание матрицы полиномиальных признаков
    X = np.column_stack([x**i for i in range(1, degree+1)])
    
    # Вычисление коэффициентов регрессии
    coef = np.linalg.inv(X.T @ X) @ X.T @ y
    
    # Вычисление предсказаний
    y_pred = X @ coef
    
    return coef, y_pred
